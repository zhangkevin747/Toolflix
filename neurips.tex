\section{Method}
\label{sec:method}
%% ============================================================

% Walk through the full pipeline.

\subsection{Two-stage retrieval}

% Stage 1: sentence-transformer embeddings (all-MiniLM-L6-v2) + cosine
% similarity. Hybrid with BM25 for recall. Returns top-N (~10) candidates.
% Job is high recall, not precision.

\subsection{Wide \& deep reranker}

% Stage 2: the learned model. Describe the shared deep trunk (query/description
% projections, interaction features, learned endpoint embeddings) and the
% two heads.

\paragraph{Relevance head.}
% Deep features only. Learns corrected semantic matching via learned endpoint
% embeddings that diverge from descriptions. Training signal: agent selection
% as pairwise preference (selected > not selected). Captures the
% description-reality gap.

\paragraph{Quality head.}
% Wide features (success rate, usage count, co-occurrence, category match) +
% deep features. Training signal: execution outcome as pairwise margin
% (succeeded > others, failed < others). Captures underdifferentiation.
% This is the head that beats static retrieval.

\paragraph{Score combination.}
% Alpha-blended: alpha * relevance + (1 - alpha) * quality.
% Alpha decays with feedback count (sigmoid schedule).
% UCB exploration bonus for underexplored endpoints.

\subsection{Online learning}

% Wide features update after every task (O(1)). Deep model retrains on
% replay buffer every 50 examples. Three-phase training: relevance head,
% quality head, joint fine-tune. Describe the pairwise margin loss with
% hard negative mining (same-server, both-high-similarity pairs).
